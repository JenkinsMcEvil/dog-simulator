class Dog
    def initialize(name, breed, status)
        @name = name
        @breed = breed
        @status = status
        case breed
        when "German Shepherd"
           @size = "large"
           @status = "awake"
        when "Labrador"
           @size = "medium"
           @status = "awake"
        when "Shiba Inu"
           @size = "medium"
           @status = "awake"
        when "Beagle"
           @size = "small"
           @status = "awake"
        when "Pomeranian"
           @size = "toy"
           @status = "awake"
        when "Chihuahua"
           @size = "toy"
           @status = "awake"
        when "Golden Retriever"
           @size = "medium"
           @status = "awake"
        when "Bulldog"
           @size = "small"
           @status = "awake"
        when "Siberian Husky"
           @size = "large"
           @status = "awake"
        when "Rottweiler"
           @size = "extra large"
           @status = "awake"
        when "Irish Wolfhound"
           @size = "extra large"
           @status = "awake"
        when "Pug"
           @size = "small"
           @status = "awake"
        when "Jack Russell Terrier"
           @size = "small"
           @status = "awake"
        when "Corgi"
           @size = "small"
           @status = "awake"
        when "Dachshund"
           @size = "small"
           @status = "awake"
        when "Sheepdog"
           @size = "large"
           @status = "awake"
        when "Basset Hound"
           @size = "small"
           @status = "awake"
        when "Samoyed"
           @size = "small"
           @status = "awake"
        when "Yorkshire Terrier"
           @size = "small"   
           @status = "awake"
        when "Poodle"
           @size = "small"
           @status = "awake"
        when "Toy Poodle"
           @size = "toy"
           @status = "awake"
      
  
        else
           @size = "indeterminate"
           @status = "unknown"
        end
    end
    def bark
        if @size == "small" or @size == "toy"
            puts "Yip!"
        else
            puts "Woof!"
        
        end
    end
end
