class Dog
    def initialize(name, breed)
        @name = name
        @breed = breed
        case breed
        when "German Shepherd"
           @size = "large"
        when "Labrador"
           @size = "medium"
        when "Shiba Inu"
           @size = "medium"
        when "Beagle"
           @size = "small"
        when "Pomeranian"
           @size = "toy"
        when "Chihuahua"
           @size = "toy"
        when "Golden Retriever"
           @size = "medium"
        when "Bulldog"
           @size = "small"
        when "Siberian Husky"
           @size = "large"
        when "Rottweiler"
           @size = "extra large"
        when "Pug"
           @size = "small"
        when "Jack Russell Terrier"
           @size = "small"
        when "Corgi"
           @size = "small"
        when "Dachshund"
           @size = "small"
        when "Sheepdog"
           @size = "large"
        when "Basset Hound"
           @size = "small"
        when "Samoyed"
           @size = "small"
        when "Yorkshire Terrier"
           @size = "small"   
           
        else
           @size = "indeterminate"
        end
    end
    def bark
        if @size == "small" or @size == "toy"
            puts "Yip!"
        else
            puts "Woof!"
        end
    end
end
